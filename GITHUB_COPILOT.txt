class EnsembleHealthMonitor:
    def track_feature_drift(self, recent_features, baseline_features):
        """Monitor SHAP value changes - trigger retraining if drift >20%"""
        shap_drift = abs(recent_features.mean() - baseline_features.mean())
        if shap_drift > 0.2:
            logger.warning(f"DRIFT DETECTED: {shap_drift:.2%}")
            return self.trigger_retraining()
        return None
    
    def cross_timeframe_voting(self, predictions_1h, predictions_4h, predictions_daily, volatility):
        """Weight votes based on market regime"""
        if volatility > 1.5:  # High vol → favor short-term
            weights = [0.5, 0.3, 0.2]
        else:  # Normal → trust all equally
            weights = [0.33, 0.33, 0.34]
        return np.average([predictions_1h, predictions_4h, predictions_daily], weights=weights)

class OrderManager:
    async def execute_with_safeguards(self, signal: BrainDecision):
        """Pre-execution risk gates"""
        # Risk Gate 1: Max position size
        if signal.quantity > self.max_position_size:
            signal.quantity = self.max_position_size
            logger.warning(f"Position size capped to {self.max_position_size}")
        
        # Risk Gate 2: Circuit breaker
        if self.portfolio.drawdown > 15%:
            logger.critical("Circuit breaker ACTIVATED")
            return None
        
        # Risk Gate 3: Liquidity check
        avg_volume = await self.data_fetcher.get_adv(signal.symbol)
        if signal.quantity > avg_volume * 0.1:
            logger.warning("Order >10% ADV - reducing size")
            signal.quantity = int(avg_volume * 0.1)
        
        return await self.broker_adapter.execute_order(signal)

class MarketMicrostructure:
    def apply_slippage(self, order_price: float, order_size: int, 
                      market_depth: dict, volatility: float) -> tuple:
        """Model realistic slippage based on market depth"""
        base_spread = market_depth['bid_ask_spread']
        volume_impact = order_size / market_depth['total_volume_10min'] * 0.05
        vol_multiplier = 1 + (volatility - 0.15) / 0.15 if volatility > 0.15 else 1
        
        total_slippage_bps = (base_spread + volume_impact) * vol_multiplier * 10000
        return order_price * (1 + total_slippage_bps / 10000), total_slippage_bps

#!/usr/bin/env python3
"""
Pre-push validation script
Runs all checks that GitHub Actions will run
"""

import subprocess
import sys
from pathlib import Path

class Colors:
    GREEN = '\033[92m'
    RED = '\033[91m'
    YELLOW = '\033[93m'
    BLUE = '\033[94m'
    RESET = '\033[0m'

def run_cmd(cmd, description):
    """Run command and report status"""
    print(f"\n{'='*70}")
    print(f"{Colors.BLUE}🔍 {description}...{Colors.RESET}")
    print(f"{'='*70}")
    
    result = subprocess.run(cmd, shell=True, capture_output=False)
    if result.returncode != 0:
        print(f"{Colors.RED}❌ FAILED: {description}{Colors.RESET}")
        return False
    print(f"{Colors.GREEN}✅ PASSED: {description}{Colors.RESET}")
    return True

def main():
    """Run all validation checks"""
    checks = [
        ("black --check . --exclude 'venv|.git|models'", 
         "Black code formatting"),
        ("isort . --check-only", 
         "isort import sorting"),
        ("flake8 . --max-line-length=100 --exclude=venv", 
         "Flake8 style check"),
        ("pylint agents/ brain/ backtesting/ data/ trading/ config/ executor/ --disable=missing-docstring,too-many-arguments --fail-under=7.0", 
         "Pylint linting"),
        ("bandit -r . -ll --exclude venv", 
         "Bandit security scan"),
        ("pytest tests/ -v --cov=. --cov-report=term-missing", 
         "Pytest + Coverage"),
    ]
    
    print(f"\n{Colors.YELLOW}{'='*70}{Colors.RESET}")
    print(f"{Colors.YELLOW}🚀 MARKET HAWK MVP - PRE-PUSH VALIDATION{Colors.RESET}")
    print(f"{Colors.YELLOW}{'='*70}{Colors.RESET}")
    
    results = []
    for cmd, desc in checks:
        results.append(run_cmd(cmd, desc))
    
    print(f"\n{Colors.BLUE}{'='*70}{Colors.RESET}")
    print(f"{Colors.BLUE}📊 SUMMARY{Colors.RESET}")
    print(f"{Colors.BLUE}{'='*70}{Colors.RESET}")
    
    passed = sum(results)
    total = len(results)
    
    for i, (_, desc) in enumerate(checks):
        status = f"{Colors.GREEN}✅{Colors.RESET}" if results[i] else f"{Colors.RED}❌{Colors.RESET}"
        print(f"{status} {desc}")
    
    print(f"\n{Colors.BLUE}Passed: {passed}/{total} checks{Colors.RESET}\n")
    
    if all(results):
        print(f"{Colors.GREEN}✅ All checks passed! Ready to push.{Colors.RESET}\n")
        return 0
    else:
        print(f"{Colors.RED}❌ Some checks failed. Fix and retry.{Colors.RESET}\n")
        return 1

if __name__ == "__main__":
    sys.exit(main())

#!/usr/bin/env python3
"""Pre-push validation script"""

import subprocess
import sys

def run_check(cmd, description):
    print(f"\n🔍 {description}...")
    result = subprocess.run(cmd, shell=True)
    return result.returncode == 0

checks = [
    ("black --check . --exclude 'venv|.git|models'", "Black formatting"),
    ("pylint agents/ brain/ backtesting/ --fail-under=7.0", "Pylint"),
    ("bandit -r . -ll --exclude venv", "Bandit security"),
    ("pytest tests/ -v --cov=.", "Pytest + Coverage"),
]

results = [run_check(cmd, desc) for cmd, desc in checks]

print(f"\n{'='*60}")
print(f"Passed: {sum(results)}/{len(results)} checks")
if all(results):
    print("✅ Ready to push!")
    sys.exit(0)
else:
    print("❌ Fix issues and retry")
    sys.exit(1)
