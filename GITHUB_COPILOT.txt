class EnsembleHealthMonitor:
    def track_feature_drift(self, recent_features, baseline_features):
        """Monitor SHAP value changes - trigger retraining if drift >20%"""
        shap_drift = abs(recent_features.mean() - baseline_features.mean())
        if shap_drift > 0.2:
            logger.warning(f"DRIFT DETECTED: {shap_drift:.2%}")
            return self.trigger_retraining()
        return None
    
    def cross_timeframe_voting(self, predictions_1h, predictions_4h, predictions_daily, volatility):
        """Weight votes based on market regime"""
        if volatility > 1.5:  # High vol → favor short-term
            weights = [0.5, 0.3, 0.2]
        else:  # Normal → trust all equally
            weights = [0.33, 0.33, 0.34]
        return np.average([predictions_1h, predictions_4h, predictions_daily], weights=weights)

class OrderManager:
    async def execute_with_safeguards(self, signal: BrainDecision):
        """Pre-execution risk gates"""
        # Risk Gate 1: Max position size
        if signal.quantity > self.max_position_size:
            signal.quantity = self.max_position_size
            logger.warning(f"Position size capped to {self.max_position_size}")
        
        # Risk Gate 2: Circuit breaker
        if self.portfolio.drawdown > 15%:
            logger.critical("Circuit breaker ACTIVATED")
            return None
        
        # Risk Gate 3: Liquidity check
        avg_volume = await self.data_fetcher.get_adv(signal.symbol)
        if signal.quantity > avg_volume * 0.1:
            logger.warning("Order >10% ADV - reducing size")
            signal.quantity = int(avg_volume * 0.1)
        
        return await self.broker_adapter.execute_order(signal)

class MarketMicrostructure:
    def apply_slippage(self, order_price: float, order_size: int, 
                      market_depth: dict, volatility: float) -> tuple:
        """Model realistic slippage based on market depth"""
        base_spread = market_depth['bid_ask_spread']
        volume_impact = order_size / market_depth['total_volume_10min'] * 0.05
        vol_multiplier = 1 + (volatility - 0.15) / 0.15 if volatility > 0.15 else 1
        
        total_slippage_bps = (base_spread + volume_impact) * vol_multiplier * 10000
        return order_price * (1 + total_slippage_bps / 10000), total_slippage_bps

#!/usr/bin/env python3
"""
Pre-push validation script
Runs all checks that GitHub Actions will run
"""

import subprocess
import sys
from pathlib import Path

class Colors:
    GREEN = '\033[92m'
    RED = '\033[91m'
    YELLOW = '\033[93m'
    BLUE = '\033[94m'
    RESET = '\033[0m'

def run_cmd(cmd, description):
    """Run command and report status"""
    print(f"\n{'='*70}")
    print(f"{Colors.BLUE}🔍 {description}...{Colors.RESET}")
    print(f"{'='*70}")
    
    result = subprocess.run(cmd, shell=True, capture_output=False)
    if result.returncode != 0:
        print(f"{Colors.RED}❌ FAILED: {description}{Colors.RESET}")
        return False
    print(f"{Colors.GREEN}✅ PASSED: {description}{Colors.RESET}")
    return True

def main():
    """Run all validation checks"""
    checks = [
        ("black --check . --exclude 'venv|.git|models'", 
         "Black code formatting"),
        ("isort . --check-only", 
         "isort import sorting"),
        ("flake8 . --max-line-length=100 --exclude=venv", 
         "Flake8 style check"),
        ("pylint agents/ brain/ backtesting/ data/ trading/ config/ executor/ --disable=missing-docstring,too-many-arguments --fail-under=7.0", 
         "Pylint linting"),
        ("bandit -r . -ll --exclude venv", 
         "Bandit security scan"),
        ("pytest tests/ -v --cov=. --cov-report=term-missing", 
         "Pytest + Coverage"),
    ]
    
    print(f"\n{Colors.YELLOW}{'='*70}{Colors.RESET}")
    print(f"{Colors.YELLOW}🚀 MARKET HAWK MVP - PRE-PUSH VALIDATION{Colors.RESET}")
    print(f"{Colors.YELLOW}{'='*70}{Colors.RESET}")
    
    results = []
    for cmd, desc in checks:
        results.append(run_cmd(cmd, desc))
    
    print(f"\n{Colors.BLUE}{'='*70}{Colors.RESET}")
    print(f"{Colors.BLUE}📊 SUMMARY{Colors.RESET}")
    print(f"{Colors.BLUE}{'='*70}{Colors.RESET}")
    
    passed = sum(results)
    total = len(results)
    
    for i, (_, desc) in enumerate(checks):
        status = f"{Colors.GREEN}✅{Colors.RESET}" if results[i] else f"{Colors.RED}❌{Colors.RESET}"
        print(f"{status} {desc}")
    
    print(f"\n{Colors.BLUE}Passed: {passed}/{total} checks{Colors.RESET}\n")
    
    if all(results):
        print(f"{Colors.GREEN}✅ All checks passed! Ready to push.{Colors.RESET}\n")
        return 0
    else:
        print(f"{Colors.RED}❌ Some checks failed. Fix and retry.{Colors.RESET}\n")
        return 1

if __name__ == "__main__":
    sys.exit(main())

#!/usr/bin/env python3
"""Pre-push validation script"""

import subprocess
import sys

def run_check(cmd, description):
    print(f"\n🔍 {description}...")
    result = subprocess.run(cmd, shell=True)
    return result.returncode == 0

checks = [
    ("black --check . --exclude 'venv|.git|models'", "Black formatting"),
    ("pylint agents/ brain/ backtesting/ --fail-under=7.0", "Pylint"),
    ("bandit -r . -ll --exclude venv", "Bandit security"),
    ("pytest tests/ -v --cov=.", "Pytest + Coverage"),
]

results = [run_check(cmd, desc) for cmd, desc in checks]

print(f"\n{'='*60}")
print(f"Passed: {sum(results)}/{len(results)} checks")
if all(results):
    print("✅ Ready to push!")
    sys.exit(0)
else:
    print("❌ Fix issues and retry")
    sys.exit(1)

#!/usr/bin/env python3
"""
PAUSABLE Video Frame Extraction
Save progress, pause anytime, resume later
"""

import cv2
import json
import pickle
from pathlib import Path
from datetime import datetime
import logging
import os
import signal
import sys

logging.basicConfig(level=logging.INFO, format='%(asctime)s | %(levelname)s | %(message)s')
logger = logging.getLogger(__name__)

class PausableVideoFrameExtractor:
    def __init__(self):
        self.timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.j_drive = Path('/mnt/j/E-Books/.....Trading Database')
        self.output_dir = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extracted_content/video_frames')
        self.output_dir.mkdir(parents=True, exist_ok=True)
        
        # CHECKPOINT file
        self.checkpoint = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extraction_checkpoint.pkl')
        self.progress_file = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extraction_progress.json')
        
        self.scene_change_threshold = 20.0
        self.max_videos = None
        
        # Load checkpoint if exists
        self.processed_videos = set()
        self.total_frames_extracted = 0
        self.load_checkpoint()
        
        # Handle interruption
        signal.signal(signal.SIGINT, self.handle_interrupt)

    def handle_interrupt(self, signum, frame):
        """Handle Ctrl+C gracefully"""
        logger.info("\n⏸️  PAUSING... Saving checkpoint...")
        self.save_checkpoint()
        logger.info("✅ Checkpoint saved! Resume later with: python scripts/pausable_video_extraction.py")
        sys.exit(0)

    def save_checkpoint(self):
        """Save progress to resume later"""
        with open(self.checkpoint, 'wb') as f:
            pickle.dump({
                'processed': self.processed_videos,
                'total_frames': self.total_frames_extracted,
                'timestamp': datetime.now().isoformat()
            }, f)
        
        # Also save JSON for inspection
        with open(self.progress_file, 'w') as f:
            json.dump({
                'processed_count': len(self.processed_videos),
                'total_frames': self.total_frames_extracted,
                'last_saved': datetime.now().isoformat(),
                'checkpoint_file': str(self.checkpoint)
            }, f, indent=2)
        
        logger.info(f"✅ Checkpoint saved! Processed: {len(self.processed_videos)} videos, {self.total_frames_extracted} frames")

    def load_checkpoint(self):
        """Load previous progress"""
        if self.checkpoint.exists():
            with open(self.checkpoint, 'rb') as f:
                data = pickle.load(f)
                self.processed_videos = data['processed']
                self.total_frames_extracted = data['total_frames']
                logger.info(f"📂 Checkpoint loaded! Resuming from {len(self.processed_videos)} videos")
                logger.info(f"   Last saved: {data['timestamp']}")
        else:
            logger.info("🆕 Fresh start - no checkpoint found")

    def extract_frames(self):
        """Main extraction with pause/resume"""
        mp4_files = sorted(list(self.j_drive.rglob('*.mp4')))
        logger.info(f"Found {len(mp4_files)} MP4 files total")
        logger.info(f"Already processed: {len(self.processed_videos)} videos")
        logger.info(f"Remaining: {len(mp4_files) - len(self.processed_videos)} videos\n")
        
        remaining = [f for f in mp4_files if f.name not in self.processed_videos]
        
        for idx, video_file in enumerate(remaining[:self.max_videos], 1):
            try:
                current_pos = len(self.processed_videos) + idx
                logger.info(f"[{current_pos}/{len(mp4_files)}] Processing: {video_file.name}")
                
                cap = cv2.VideoCapture(str(video_file))
                if not cap.isOpened():
                    logger.warning(f"⚠️ Cannot open: {video_file.name}")
                    self.processed_videos.add(video_file.name)
                    continue
                
                fps = cap.get(cv2.CAP_PROP_FPS)
                frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
                duration = frame_count / fps if fps > 0 else 0
                
                prev_frame = None
                key_frames = []
                frame_idx = 0
                
                while True:
                    ret, frame = cap.read()
                    if not ret:
                        break
                    
                    if prev_frame is not None:
                        diff = cv2.absdiff(frame, prev_frame)
                        change_percent = (diff.mean() / 255.0) * 100
                        
                        if change_percent > self.scene_change_threshold:
                            key_frames.append(frame_idx)
                    
                    prev_frame = frame.copy()
                    frame_idx += 1
                
                cap.release()
                
                # Save key frames
                output_subdir = self.output_dir / video_file.stem
                output_subdir.mkdir(parents=True, exist_ok=True)
                
                frame_count = 0
                cap = cv2.VideoCapture(str(video_file))
                for frame_num in key_frames:
                    cap.set(cv2.CAP_PROP_POS_FRAMES, frame_num)
                    ret, frame = cap.read()
                    if ret:
                        cv2.imwrite(str(output_subdir / f"frame_{frame_count:04d}.jpg"), frame)
                        frame_count += 1
                
                cap.release()
                
                self.processed_videos.add(video_file.name)
                self.total_frames_extracted += len(key_frames)
                
                logger.info(f"✅ Detected {len(key_frames)} key frames (duration: {duration:.1f}s)")
                
                # Save checkpoint every 5 videos
                if len(self.processed_videos) % 5 == 0:
                    self.save_checkpoint()
                    
            except Exception as e:
                logger.error(f"❌ Error processing {video_file.name}: {e}")
                continue
        
        # Final save
        self.save_checkpoint()
        logger.info(f"\n{'='*90}")
        logger.info(f"✅ EXTRACTION COMPLETE!")
        logger.info(f"Total videos processed: {len(self.processed_videos)}")
        logger.info(f"Total key frames extracted: {self.total_frames_extracted}")
        logger.info(f"{'='*90}")

if __name__ == '__main__':
    extractor = PausableVideoFrameExtractor()
    extractor.extract_frames()

#!/usr/bin/env python3
"""
PAUSABLE Video Frame Extraction
Save progress, pause anytime, resume later
"""

import cv2
import json
import subprocess
from pathlib import Path
from datetime import datetime
import logging
import os
import pickle

logging.basicConfig(level=logging.INFO, format='%(asctime)s | %(levelname)s | %(message)s')
logger = logging.getLogger(__name__)

class PausableVideoFrameExtractor:
    def __init__(self):
        self.timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.j_drive = Path('/mnt/j/E-Books/.....Trading Database')
        self.output_dir = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extracted_content/video_frames')
        self.output_dir.mkdir(parents=True, exist_ok=True)
        
        # CHECKPOINT file
        self.checkpoint = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extraction_checkpoint.pkl')
        self.progress_file = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extraction_progress.json')
        
        self.scene_change_threshold = 20.0
        self.max_videos = None  # Process all
        
        # Load checkpoint if exists
        self.processed_videos = set()
        self.total_frames_extracted = 0
        self.load_checkpoint()

    def save_checkpoint(self):
        """Save progress to resume later"""
        with open(self.checkpoint, 'wb') as f:
            pickle.dump({
                'processed': self.processed_videos,
                'total_frames': self.total_frames_extracted,
                'timestamp': datetime.now().isoformat()
            }, f)
        logger.info(f"✅ Checkpoint saved! Processed: {len(self.processed_videos)} videos")

    def load_checkpoint(self):
        """Load previous progress"""
        if self.checkpoint.exists():
            with open(self.checkpoint, 'rb') as f:
                data = pickle.load(f)
                self.processed_videos = data['processed']
                self.total_frames_extracted = data['total_frames']
                logger.info(f"📂 Checkpoint loaded! Resuming from {len(self.processed_videos)} videos")
                logger.info(f"   Last saved: {data['timestamp']}")
        else:
            logger.info("🆕 Fresh start - no checkpoint found")

    def extract_frames(self):
        """Main extraction with pause/resume"""
        mp4_files = sorted(list(self.j_drive.rglob('*.mp4')))
        logger.info(f"Found {len(mp4_files)} MP4 files total")
        logger.info(f"Already processed: {len(self.processed_videos)} videos")
        logger.info(f"Remaining: {len(mp4_files) - len(self.processed_videos)} videos")
        
        remaining = [f for f in mp4_files if f.name not in self.processed_videos]
        
        for idx, video_file in enumerate(remaining[:self.max_videos], 1):
            try:
                logger.info(f"\n[{len(self.processed_videos) + idx}/{len(mp4_files)}] Processing: {video_file.name}")
                
                cap = cv2.VideoCapture(str(video_file))
                if not cap.isOpened():
                    logger.warning(f"⚠️ Cannot open: {video_file.name}")
                    continue
                
                fps = cap.get(cv2.CAP_PROP_FPS)
                frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
                duration = frame_count / fps if fps > 0 else 0
                
                prev_frame = None
                key_frames = []
                frame_idx = 0
                
                while True:
                    ret, frame = cap.read()
                    if not ret:
                        break
                    
                    if prev_frame is not None:
                        diff = cv2.absdiff(frame, prev_frame)
                        change_percent = (diff.mean() / 255.0) * 100
                        
                        if change_percent > self.scene_change_threshold:
                            key_frames.append(frame_idx)
                    
                    prev_frame = frame.copy()
                    frame_idx += 1
                
                cap.release()
                
                # Save key frames
                output_subdir = self.output_dir / video_file.stem
                output_subdir.mkdir(parents=True, exist_ok=True)
                
                frame_count = 0
                cap = cv2.VideoCapture(str(video_file))
                for frame_num in key_frames:
                    cap.set(cv2.CAP_PROP_POS_FRAMES, frame_num)
                    ret, frame = cap.read()
                    if ret:
                        cv2.imwrite(str(output_subdir / f"frame_{frame_count:04d}.jpg"), frame)
                        frame_count += 1
                
                cap.release()
                
                self.processed_videos.add(video_file.name)
                self.total_frames_extracted += len(key_frames)
                
                logger.info(f"✅ Detected {len(key_frames)} key frames (duration: {duration:.1f}s, total: {frame_count})")
                
                # Save checkpoint every 5 videos
                if len(self.processed_videos) % 5 == 0:
                    self.save_checkpoint()
                    
            except Exception as e:
                logger.error(f"❌ Error processing {video_file.name}: {e}")
                continue
        
        # Final save
        self.save_checkpoint()
        logger.info(f"\n{'='*90}")
        logger.info(f"✅ EXTRACTION COMPLETE!")
        logger.info(f"Total videos processed: {len(self.processed_videos)}")
        logger.info(f"Total key frames extracted: {self.total_frames_extracted}")
        logger.info(f"{'='*90}")

if __name__ == '__main__':
    extractor = PausableVideoFrameExtractor()
    extractor.extract_frames()

import cv2
import json
import pickle
from pathlib import Path
from datetime import datetime
import logging
import os
import signal
import sys

logging.basicConfig(level=logging.INFO, format='%(asctime)s | %(levelname)s | %(message)s')
logger = logging.getLogger(__name__)

class PausableVideoFrameExtractor:
    def __init__(self):
        self.timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.j_drive = Path('/mnt/j/E-Books/.....Trading Database')
        self.output_dir = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extracted_content/video_frames')

        # CHECKPOINT file
        self.checkpoint = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extraction_checkpoint.pkl')
        self.progress_file = Path('/mnt/k/_DEV_MVP_2026/Market_Hawk_3/extraction_progress.json')
        
        self.scene_change_threshold = 20.0
        self.max_videos = None
        
        # Load checkpoint if exists
        self.processed_videos = set()
        self.total_frames_extracted = 0
        self.load_checkpoint()
        
        # Handle interruption
        signal.signal(signal.SIGINT, self.handle_interrupt)

    def handle_interrupt(self, signum, frame):
        """Handle Ctrl+C gracefully"""
        logger.info("\n⏸️  PAUSING... Saving checkpoint...")
        self.save_checkpoint()
        logger.info("✅ Checkpoint saved! Resume later with: python scripts/pausable_video_extraction.py")
        sys.exit(0)

    def save_checkpoint(self):
        """Save progress to resume later"""
        with open(self.checkpoint, 'wb') as f:
            pickle.dump({
                'processed': self.processed_videos,
                'total_frames': self.total_frames_extracted,
                'timestamp': datetime.now().isoformat()
            }, f)
        
        # Also save JSON for inspection
        with open(self.progress_file, 'w') as f:
            json.dump({
                'processed_count': len(self.processed_videos),
                'total_frames': self.total_frames_extracted,
                'last_saved': datetime.now().isoformat(),
                'checkpoint_file': str(self.checkpoint)
            }, f, indent=2)
        
        logger.info(f"✅ Checkpoint saved! Processed: {len(self.processed_videos)} videos, {self.total_frames_extracted} frames")

    def load_checkpoint(self):
        """Load previous progress"""
        if self.checkpoint.exists():
            with open(self.checkpoint, 'rb') as f:
                data = pickle.load(f)
                self.processed_videos = data['processed']
                self.total_frames_extracted = data['total_frames']
                logger.info(f"📂 Checkpoint loaded! Resuming from {len(self.processed_videos)} videos")
                logger.info(f"   Last saved: {data['timestamp']}")
        else:
            logger.info("🆕 Fresh start - no checkpoint found")

    def extract_frames(self):
        """Main extraction with pause/resume"""
        mp4_files = sorted(list(self.j_drive.rglob('*.mp4')))
        logger.info(f"Found {len(mp4_files)} MP4 files total")
        logger.info(f"Already processed: {len(self.processed_videos)} videos")
        logger.info(f"Remaining: {len(mp4_files) - len(self.processed_videos)} videos\n")
        
        remaining = [f for f in mp4_files if f.name not in self.processed_videos]
        
        for idx, video_file in enumerate(remaining[:self.max_videos], 1):
            try:
                current_pos = len(self.processed_videos) + idx
                logger.info(f"[{current_pos}/{len(mp4_files)}] Processing: {video_file.name}")
                
                cap = cv2.VideoCapture(str(video_file))
                if not cap.isOpened():
                    logger.warning(f"⚠️ Cannot open: {video_file.name}")
                    self.processed_videos.add(video_file.name)
                    continue
                
                fps = cap.get(cv2.CAP_PROP_FPS)
                frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
                duration = frame_count / fps if fps > 0 else 0
                
                prev_frame = None
                key_frames = []
                frame_idx = 0
                
                while True:
                    ret, frame = cap.read()
                    if not ret:
                        break
                    
                    if prev_frame is not None:
                        diff = cv2.absdiff(frame, prev_frame)
                        change_percent = (diff.mean() / 255.0) * 100
                        
                        if change_percent > self.scene_change_threshold:
                            key_frames.append(frame_idx)
                    
                    prev_frame = frame.copy()
                    frame_idx += 1
                
                cap.release()
                
                # Save key frames
                output_subdir = self.output_dir / video_file.stem
                output_subdir.mkdir(parents=True, exist_ok=True)
                
                frame_count = 0
                cap = cv2.VideoCapture(str(video_file))
                for frame_num in key_frames:
                    cap.set(cv2.CAP_PROP_POS_FRAMES, frame_num)
                    ret, frame = cap.read()
                    if ret:
                        cv2.imwrite(str(output_subdir / f"frame_{frame_count:04d}.jpg"), frame)
                        frame_count += 1
                
                cap.release()
                
                self.processed_videos.add(video_file.name)
                self.total_frames_extracted += len(key_frames)
                
                logger.info(f"✅ Detected {len(key_frames)} key frames (duration: {duration:.1f}s)")
                
                # Save checkpoint every 5 videos
                if len(self.processed_videos) % 5 == 0:
                    self.save_checkpoint()
                    
            except Exception as e:
                logger.error(f"❌ Error processing {video_file.name}: {e}")
                continue
        
        # Final save
        self.save_checkpoint()
        logger.info(f"\n{'='*90}")
        logger.info(f"✅ EXTRACTION COMPLETE!")
        logger.info(f"Total videos processed: {len(self.processed_videos)}")
        logger.info(f"Total key frames extracted: {self.total_frames_extracted}")
        logger.info(f"{'='*90}")

if __name__ == '__main__':
    extractor = PausableVideoFrameExtractor()
    extractor.extract_frames()
